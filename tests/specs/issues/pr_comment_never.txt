== Case 1: func with multi-line args (never) ==
func({
a: 1,
b: 2,
}, {
c: 3
})

[expect]
func({
    a: 1,
    b: 2,
}, {
    c: 3,
});

== Case 2: func with mixed args (never) ==
func(1, 2, {
c: 3
})

[expect]
func(1, 2, {
    c: 3,
});

== Case 3: func with middle multi-line arg (never) ==
func(1, {
c: 3
}, 3)

[expect]
func(1, {
    c: 3,
}, 3);
